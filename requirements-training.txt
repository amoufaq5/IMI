# =============================================================================
# requirements-training.txt — Mixtral 8x7B Medical Training
# Covers both:
#   (A) Foundation Training  — full fine-tuning, 8× A100 80GB, DeepSpeed ZeRO-3
#   (B) Adapter Training     — QLoRA, single A100 80GB, no DeepSpeed
# =============================================================================
#
# WARNING: DO NOT run `pip install -r requirements-training.txt` directly.
# Packages must be installed in strict order to avoid breaking CUDA/torch.
# Use scripts/install_training.sh for a safe, ordered installation.
#
# Compatible with: CUDA 12.4 + PyTorch 2.4.1 (RunPod default), CUDA 12.1 + PyTorch 2.2.x
# =============================================================================

# ── STAGE 1: PyTorch — installed via --index-url (see install_training.sh) ───
# NOT listed here — install_training.sh handles torch separately because:
#   1. RunPod pre-installs a matched torch+torchaudio+torchvision set.
#      Overwriting torch alone breaks torchaudio/torchvision.
#   2. The script detects the existing version and skips reinstall if CUDA works.
#
# Default versions installed on a fresh system:
#   CUDA 12.4 (RunPod default):  torch==2.4.1+cu124
#   CUDA 12.1:                   torch==2.2.0+cu121
#   CUDA 11.8:                   torch==2.2.0+cu118

# ── STAGE 2: Core Training Stack ──────────────────────────────────────────────
# Install AFTER torch: `python -c "import torch; print(torch.cuda.is_available())"`

transformers==4.44.2          # Mixtral model support, SFTTrainer compat (torch 2.4.x compat)
accelerate==0.34.2            # Multi-GPU / mixed precision wrapper (torch 2.4.x compat)
peft==0.12.0                  # LoRA adapters — ADAPTER TRAINING only, not foundation
trl==0.9.6                    # SFTTrainer — DO NOT upgrade past 0.9.x without testing
bitsandbytes==0.44.0          # 4-bit NF4 (QLoRA) — ADAPTER TRAINING only; 0.44.0 for torch 2.4.x
datasets==2.21.0              # HuggingFace datasets loader
safetensors==0.4.5            # Fast model checkpoint format

# ── STAGE 3: Tokenization & Utilities ─────────────────────────────────────────
sentencepiece==0.1.99         # Mixtral tokenizer (required)
tokenizers==0.19.1            # Fast tokenizers
einops==0.8.0                 # Tensor rearrangement ops used by Mixtral attention
numpy==1.26.4                 # Must stay <2.0 for PyTorch 2.2 compatibility
pandas==2.2.1                 # Data manipulation
tqdm>=4.65.0                  # Progress bars
pyyaml>=6.0.1                 # Config file support
packaging>=23.0               # Required by accelerate and deepspeed

# ── STAGE 4: DeepSpeed — FOUNDATION TRAINING only (8× A100) ──────────────────
# Required for ZeRO Stage 3 sharding across 8 GPUs.
# NOT needed for adapter training (single GPU QLoRA).
#
# DeepSpeed builds CUDA kernels on install. Install AFTER torch is verified:
#   pip install deepspeed==0.14.0
#
# Requires: gcc, g++, ninja, CUDA toolkit headers matching your driver.
# On RunPod / Lambda these are pre-installed. Verify with:
#   ds_report   (run after install to check which ops compiled successfully)
#
# If deepspeed install fails:
#   apt-get install -y ninja-build build-essential
#   DS_BUILD_CPU_ADAM=1 pip install deepspeed==0.14.0
ninja>=1.11.1                 # C++ build tool — required for DeepSpeed CUDA kernels

# ── STAGE 5: Flash Attention — install LAST, build from source ───────────────
# Speeds up training ~30–50% on A100/H100.
# Must exactly match torch + CUDA version. Training works without it.
#
# Install with:
#   pip install flash-attn==2.5.9 --no-build-isolation
#
# If build fails, re-run install_training.sh with --skip-flash
